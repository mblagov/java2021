package com.mironova.task1;

import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;
import java.util.Scanner;

public class ATM {

    public static void main(String[] args) {
        System.out.println("Input total sum:");
        long sumForExchange = 0;
        int flag = 0;
        String str;
        Scanner scanner = new Scanner(System.in);
        if (scanner.hasNextLong()) {
            flag++;
            sumForExchange = scanner.nextLong();
        }
        else {
            System.out.println("Not an integer or too big value");
        }

        System.out.println("Input values:");
        scanner = new Scanner(System.in);
        str = scanner.nextLine();

        String[] strValues = str.split(" ");
        int n = strValues.length;
        long[] values = new long[n];
        try {
            for (int i = 0; i < n; i++) {
                values[i] = Long.parseLong(strValues[i]);
            }
            flag++;
        }
        catch (NumberFormatException e) {
            System.out.println("Not an integer");
        }
        finally {
            scanner.close();
        }

        Arrays.sort(values);
        if (flag == 2) {
            List<List<Long>> vars = exchange(values, sumForExchange, values[n - 1]);

            for (List<Long> var : vars) {
                for (Long l : var) {
                    System.out.print(l + " ");
                }
                System.out.println();
            }
        }

    }

    private static List<List<Long>> exchange(long[] values, long note, long maxValue) {
        List<List<Long>> result = new ArrayList<>();
        if (note == 0) {
            result.add(new ArrayList<>());
        } else {
            for (int i = values.length - 1; i >= 0; i--) {
                long coin = values[i];
                if (coin <= note && coin <= maxValue) {
                    for (List<Long> remain : exchange(values, note - coin, coin)) {
                        List<Long> set = new ArrayList<>();
                        set.add(coin);
                        set.addAll(remain);
                        result.add(set);
                    }
                }
            }
        }
        return result;
    }
}
